# Na VM Edge — deploy manual (sem Flux)
# Volte para a raiz do repo (os manifestos estão em edge/k8s):
cd /mnt/c/_repo/azure-iot/video-analytics-demo/azure-iot

# 1. Criar o namespace e aplicar os manifestos (namespace -> config/deploy/service)
kubectl apply -f edge/k8s/namespace.yaml
kubectl apply -f edge/k8s/video-analytics/

# 2. (uma única vez) criar o secret de pull do ACR no namespace video-analytics
#    O Deployment referencia imagePullSecrets: acr-pull-secret.
ACR_NAME=aiotdemoacr
ACR_USER=$(az acr credential show -n $ACR_NAME --query username -o tsv)
ACR_PASS=$(az acr credential show -n $ACR_NAME --query 'passwords[0].value' -o tsv)
kubectl create secret docker-registry acr-pull-secret \
  --namespace video-analytics \
  --docker-server=$ACR_NAME.azurecr.io \
  --docker-username=$ACR_USER \
  --docker-password=$ACR_PASS \
  --dry-run=client -o yaml | kubectl apply -f -

# 3. Apontar para a tag desta build e reiniciar
kubectl set image deployment/video-analytics \
  video-analytics=aiotdemoacr.azurecr.io/video-analytics:latest \
  -n video-analytics

kubectl rollout restart deployment/video-analytics -n video-analytics

# 4. Verificar 
kubectl rollout status deployment/video-analytics -n video-analytics
kubectl get pods -n video-analytics
kubectl logs -n video-analytics deployment/video-analytics -f
